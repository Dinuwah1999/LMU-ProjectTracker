Ansible VLAN Configuration

- name: Configure VLANs
  hosts: gns3_switches
  tasks:
    - name: Create VLANs
      ios_vlans:
        config:
          - name: Management
            vlan_id: 10
          - name: Production
            vlan_id: 20
          - name: DMZ
            vlan_id: 30

Docker Compose for ELK Stack

version: '3'
services:
  elasticsearch:
    image: docker.elastic.co/elasticsearch/elasticsearch:7.15.0
    ports:
      - 9200:9200
    environment:
      - discovery.type=single-node
  
  kibana:
    image: docker.elastic.co/kibana/kibana:7.15.0
    ports:
      - 5601:5601
    depends_on:
      - elasticsearch
